代码二——能实现

C代码

#include<stdio.h>
void swap(int *a,int *b)
{
	int c = *a;
	*a = *b;
	*b = c;
}
int main()
{
	int a = 1, b = 0;
	swap(&a,&b);
	printf("%d  %d", a, b);
	return 0;
}
